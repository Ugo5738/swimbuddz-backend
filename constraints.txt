# pyproject.toml declares version RANGES (e.g. fastapi>=0.109,<0.112). Without a
# lock, CI's fresh `pip install` floats to newer FastAPI/Pydantic than the built
# containers, which changes FastAPI's input/output OpenAPI schema generation and
# breaks the "Verify openapi.json is up to date" check non-deterministically.
#

# that generated the committed openapi.json), so CI and local/container envs
# install identical versions.
#

#   docker compose exec sessions-service pip freeze \
#     | grep -ivE '^(-e |swimbuddz|pip==|setuptools==|wheel==|pkg[-_]resources==)' \
#     | sort > /tmp/body && cat <header> /tmp/body > constraints.txt
